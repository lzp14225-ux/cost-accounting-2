# pricing_server_sse.py
async def call_tool(name: str, arguments: dict):
    subgraphs = arguments.get("subgraphs", [])
    
    # 按 calc_type 分组
    groups = {}
    for sg in subgraphs:
        calc_type = sg.get("calc_type", "base")  # 默认走 base
        groups.setdefault(calc_type, []).append(sg)
    
    # 分发到不同计算器
    calculators = {
        "base": price_base.calculate,
        "wire": price_wire.calculate,
        "milling": price_milling.calculate,
    }
    
    results = []
    for calc_type, items in groups.items():
        calc_func = calculators.get(calc_type, price_base.calculate)
        results.extend(await calc_func(items))
    
    return results

# pricing_agent.py
async def search_and_calculate(self, job_ids):
